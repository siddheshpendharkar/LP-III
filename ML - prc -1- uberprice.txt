import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# 1. Load Dataset
df = pd.read_csv("uber.csv")
print("Initial Shape:", df.shape)

# 2. Remove Missing Values
df.dropna(inplace=True)
print("After Removing Missing Values:", df.shape)

# 3. Calculate Distance (Haversine Formula)
def haversine(lon1, lon2, lat1, lat2):
    lon1, lon2, lat1, lat2 = map(np.radians, [lon1, lon2, lat1, lat2])
    dlon = lon2 - lon1
    dlat = lat2 - lat1
    a = np.sin(dlat/2)**2 + np.cos(lat1)*np.cos(lat2)*np.sin(dlon/2)**2
    c = 2 * np.arcsin(np.sqrt(a))
    return 6371 * c  # Earth radius in KM

df["Distance"] = haversine(df["pickup_longitude"], df["dropoff_longitude"],df["pickup_latitude"], df["dropoff_latitude"]).round(2)

# Keep only realistic values
df = df[(df["Distance"] > 0) & (df["Distance"] < 60)]
df = df[(df["fare_amount"] > 0) & (df["fare_amount"] < 500)]

print("Shape After Removing Outliers:", df.shape)

# 4. Correlation Check
corr = df[["fare_amount", "Distance"]].corr()
print("\nCorrelation Matrix:\n", corr)

plt.figure(figsize=(5,4))
sns.heatmap(corr, annot=True, cmap="coolwarm")
plt.title("Correlation Heatmap")
plt.show()

# 5. Visualization - Raw Relationship
plt.figure(figsize=(6,4))
plt.scatter(df["Distance"], df["fare_amount"], alpha=0.4)
plt.xlabel("Distance (Km)")
plt.ylabel("Fare Amount")
plt.title("Fare vs Distance")
plt.show()

# Prepare Data
X = df[["Distance"]].values
y = df["fare_amount"].values

# Train-Test Split
from sklearn.model_selection import train_test_split
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=0)

# 6. Linear Regression Model
from sklearn.linear_model import LinearRegression
lr = LinearRegression()
lr.fit(X_train, y_train)
y_pred_lr = lr.predict(X_test)

# 7. Random Forest Regressor Model
from sklearn.ensemble import RandomForestRegressor
rf = RandomForestRegressor()
rf.fit(X_train, y_train)
y_pred_rf = rf.predict(X_test)

# 8. Evaluation Metrics
from sklearn.metrics import mean_squared_error, r2_score
from sklearn.metrics import mean_absolute_error, mean_squared_error, r2_score

print("\n--- Model Evaluation ---")

# Linear Regression
mae_lr = mean_absolute_error(y_test, y_pred_lr)
mse_lr = mean_squared_error(y_test, y_pred_lr)
rmse_lr = np.sqrt(mse_lr)
r2_lr = r2_score(y_test, y_pred_lr)

print("Linear Regression:")
print("R² Score:", r2_lr)
print("MAE:", mae_lr)
print("MSE:", mse_lr)
print("RMSE:", rmse_lr)

# Random Forest
mae_rf = mean_absolute_error(y_test, y_pred_rf)
mse_rf = mean_squared_error(y_test, y_pred_rf)
rmse_rf = np.sqrt(mse_rf)
r2_rf = r2_score(y_test, y_pred_rf)

print("\nRandom Forest:")
print("R² Score:", r2_rf)
print("MAE:", mae_rf)
print("MSE:", mse_rf)
print("RMSE:", rmse_rf)

# 9. Comparison Visualization
plt.figure(figsize=(12,5))

# Linear Regression Plot
plt.subplot(1, 2, 1)
plt.scatter(X_test, y_test, label="Actual Fare", alpha=0.6)
plt.plot(X_test, y_pred_lr, color='red', label="Predicted Fare (LR)", linewidth=2)
plt.xlabel("Distance")
plt.ylabel("Fare")
plt.title("Linear Regression")
plt.legend()

# Random Forest Plot
plt.subplot(1, 2, 2)
plt.scatter(X_test, y_test, label="Actual Fare", alpha=0.6)
plt.scatter(X_test, y_pred_rf, color='green', label="Predicted Fare (RF)", alpha=0.6)
plt.xlabel("Distance")
plt.ylabel("Fare")
plt.title("Random Forest Regression")
plt.legend()

plt.tight_layout()
plt.show()
